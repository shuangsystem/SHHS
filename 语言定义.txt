-- SHHS (SHuang HaSkell)
data S a := 集合
  -- 基数可以大于ℵ0, 所以不能被折叠
data Atom := 只指代一个事物/抽象事物的纯粹的名词
  -- 这个定义只会在定义名词的时候用一下, 用于帮助理解
data N = S Atom
  -- N是名词
data AtomM := 原子陈述
newtype M
  -- 定义在Data/Set.shhs
data P a := 代词, 代指一个类型为a的事物
  -- 代词可以在不同语境绑定不同东西
data Count := 量词
data Time := 时间
type V = N -> M
type V1 = N -> V
type V2 = N -> V1
type Content = M -> M
